CREATE TABLE Consumer 
(
CID VARCHAR(30) NOT NULL,
c_name VARCHAR(30) NOT NULL, 
password VARCHAR(200) NOT NULL,
UnivID VARCHAR(30) NOT NULL,
Email VARCHAR(30) NOT NULL,
Point INT(10) NOT NULL,
salt INT(20) NOT NULL,
PRIMARY KEY (CID)
);

CREATE TABLE History
(
PID INT auto_increment NOT NULL,
Purchase_num INT NOT NULL,
HCID VARCHAR(30) NOT NULL,
HItemID INT NOT NULL,
h_status INT NOT NULL, 
PRIMARY KEY (PID),
FOREIGN KEY (HCID) REFERENCES Consumer(CID),
FOREIGN KEY (HItemID) REFERENCES Item(ItemID)
);


CREATE TABLE Item 
(
ItemID INT auto_increment NOT NULL,
ISID VARCHAR(30) NOT NULL,
img_src VARCHAR(100),
Period TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP(),
EndDate TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP(),
cur_total INT NOT NULL,
description VARCHAR(300),
is_Available INT NOT NULL,
cur_price INT NOT NULL,
i_name VARCHAR(30) NOT NULL,
Univ_ID VARCHAR(10) NOT NULL,
Discount1 INT NOT NULL,
Discount2 INT NOT NULL, 
Discount3 INT NOT NULL, 
Disc_num1 INT NOT NULL, 
Disc_num2 INT NOT NULL, 
Disc_num3 INT NOT NULL,
i_status INT NOT NULL,
PRIMARY KEY (ItemID),
FOREIGN KEY(ISID) REFERENCES Seller(SID)
);

CREATE TABLE Seller
(
SID VARCHAR(30) NOT NULL,
s_name VARCHAR(30) NOT NULL, 
CompanyID VARCHAR(30) NOT NULL,
password VARCHAR(200) NOT NULL,
Point INT(10) NOT NULL,
salt INT(20) NOT NULL,
PRIMARY KEY (SID)
);
